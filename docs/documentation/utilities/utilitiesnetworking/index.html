Instance Methods

func getConnectionType() -> String
возвращает тип соединения 2G, 3G, 4G, WIFI или NO INTERNET
func isConnection() -> Bool
проверяет есть ли данный момент соединение с wifi или сотовой связью
func isRoaming((_ result: String) -> Void)
Функция выясняет находится ли пользователь в роуменге, путем отправки запроса на https://api.bigdatacloud.net, запрос отправляет текущее местоположение и возвращает Булевое значение. Предварительно функция проверит доступ к геолокации и запросит доступ, если понадобится.
func localIP() -> [String]
Возвращает массив внутренних ip адресов устройства
func publicIP((_ result: String) -> Void)
Возвращает в замыкании публичный ip, путем отправки запроса по “https://api.ipify.org/?format=json”
func reachabilityCallback(reachable: (Bool) -> ())
проверяет есть ли данный момент соединение с wifi или сотовой связью в реальном времени через замыкание предварительно нужно включить reachabilityStartNotifier()
func reachabilityPublisher() -> AnyPublisher<String, Never>
проверяет есть ли данный момент соединение с wifi или сотовой связью в реальном времени через паблишер предварительно нужно включить reachabilityStartNotifier()
func reachabilityStartNotifier()
Начинает следить за состоянием сетевого соединения через NotificationCenter. Для обработки нужно использовать reachabilityCallback() или reachabilityPublisher()
func reachabilityStopNotifier()
Прекращает отслеживать состояние сетевого соединения
